% --- Patches for \chapter (numbered) ---

% 1. Space ABOVE "Chapter X"
\patchcmd{\@makechapterhead}
  {\vspace*{50\p@}}
  {\vspace*{-25\p@}}
  {}{}

% 2. Space BETWEEN "Chapter X" and "Title" (This is the new one!)
\patchcmd{\@makechapterhead}
  {\vskip 20\p@}     % This is the default space
  {\vskip 10\p@}      % <-- CHANGE THIS to your desired space
  {}{}

% 3. Space BELOW "Title"
\patchcmd{\@makechapterhead}
  {\vskip 40\p@}
  {\vspace*{10\p@}} % Your original replacement
  {}{}

% --- Patches for \chapter* (starred) ---
